        function showView(viewId) {
            ['view-registration', 'view-dashboard', 'view-test', 'view-module-result'].forEach(v => {
                document.getElementById(v).classList.add('hidden');
                document.getElementById(v).classList.remove('flex');
            });
            const target = document.getElementById(viewId);
            target.classList.remove('hidden');
            if (viewId === 'view-test' || viewId === 'view-module-result') target.classList.add('flex');
        }

        function handleRegistration() {
            const name = document.getElementById('reg-name').value.trim();
            const phone = document.getElementById('reg-phone').value.trim();
            
            if (name.split(' ').length < 2) {
                alert("Пожалуйста, введите полные Фамилию и Имя.");
                return;
            }
            if (phone.length < 10) {
                alert("Пожалуйста, введите корректный номер телефона.");
                return;
            }

            currentUser = { name, phone };
            localStorage.setItem('mob_user', JSON.stringify(currentUser));
            
            document.getElementById('header-user-info').classList.remove('hidden');
            document.getElementById('header-name').innerText = name;
            
            renderDashboard();
            showView('view-dashboard');
        }

            const list = document.getElementById('modules-list');
            list.innerHTML = '';
            let completedCount = 0;
            let totalScore = 0;

            MODULES.forEach(mod => {
                const res = resultsStore[mod.id];
                const card = document.createElement('div');
                card.className = "bg-white border rounded-2xl p-5 shadow-sm module-card hover:shadow-md border-slate-200 flex flex-col sm:flex-row gap-4 justify-between items-start sm:items-center";
                
                let statusBadge = `<span class="bg-slate-100 text-slate-500 text-[10px] font-bold px-2 py-1 rounded uppercase tracking-wide">Не пройдено</span>`;

                }

        function startModule(id) {
            currentModuleId = id;
            userAnswers = {};
            currentQIndex = 0;
            timeLeft = 900; // 15 minutes strict

            // Filter questions for this module, shuffle them, and pick 10
            let modQuestions = ALL_QUESTIONS.filter(q => q.m === id);
            testQuestions = modQuestions.sort(() => 0.5 - Math.random()).slice(0, 10);

            document.getElementById('test-module-badge').innerText = `Модуль ${id}`;
            renderQuestionNavDots();
            loadQuestion();
            startTimer();
            showView('view-test');
        }

        function startTimer() {
            if (timerInterval) clearInterval(timerInterval);
            timerInterval = setInterval(() => {
                timeLeft--;
                if (timeLeft <= 0) {
                    clearInterval(timerInterval);
                    alert("Время вышло! (15 минут истекли). Ваши текущие ответы сохранены.");
                    finishModuleTest();
                } else {
                    const m = Math.floor(timeLeft / 60);
                    const s = timeLeft % 60;
                    document.getElementById('timer-text').innerText = `${m}:${s.toString().padStart(2,'0')}`;
                    
                    if (timeLeft < 180) { // last 3 mins turns red and pulses
                        document.getElementById('test-timer').className = "flex items-center space-x-1.5 text-white font-bold text-sm bg-rose-500 px-3 py-1.5 rounded-xl border border-rose-600 animate-pulse";
                    } else {
                        document.getElementById('test-timer').className = "flex items-center space-x-1.5 text-rose-600 font-bold text-sm bg-rose-50 px-3 py-1.5 rounded-xl border border-rose-100";
                    }
                }
            }, 1000);
        }

        function renderQuestionNavDots() {
            const container = document.getElementById('nav-dots');
            container.innerHTML = '';
            for (let i=0; i<10; i++) {
                const dot = document.createElement('button');
                dot.id = `dot-${i}`;
                dot.onclick = () => { 
                    if (autoAdvanceTimer) { clearTimeout(autoAdvanceTimer); autoAdvanceTimer = null; }
                    currentQIndex = i; 
                    loadQuestion(); 
                };
                dot.className = `h-2 flex-grow rounded-full transition-colors ${i === currentQIndex ? 'bg-blue-600' : (userAnswers[i] !== undefined ? 'bg-blue-200' : 'bg-slate-200')}`;
                container.appendChild(dot);
            }
        }

        function loadQuestion() {
            const q = testQuestions[currentQIndex];
            if (!q) return; // Защита от выхода за пределы массива
            
            document.getElementById('test-q-current').innerText = currentQIndex + 1;
            document.getElementById('question-text').innerText = q.text;
            
            const pct = ((currentQIndex) / 10) * 100;
            document.getElementById('test-progress-bar').style.width = pct + '%';

            const grid = document.getElementById('options-grid');
            grid.innerHTML = '';
            const letters = ['А', 'Б', 'В', 'Г'];
            
            q.opts.forEach((optText, idx) => {
                const btn = document.createElement('button');
                const isSelected = userAnswers[currentQIndex] === idx;
                
                btn.className = `w-full p-4 rounded-2xl border text-left flex items-start space-x-4 transition-all duration-150 ${isSelected ? 'bg-blue-50 border-blue-500 ring-1 ring-blue-400 shadow-sm' : 'bg-white border-slate-200 hover:bg-slate-50'}`;
                btn.onclick = () => {
                    if (autoAdvanceTimer) { clearTimeout(autoAdvanceTimer); autoAdvanceTimer = null; }
                    userAnswers[currentQIndex] = idx;
                    loadQuestion(); // Моментально обновляем визуал
                    
                    if (currentQIndex < 9) {
                        autoAdvanceTimer = setTimeout(() => { 
                            autoAdvanceTimer = null;
                            currentQIndex++; 
                            loadQuestion(); 
                        }, 350);
                    }
                };
                
                btn.innerHTML = `
                    <span class="w-8 h-8 rounded-xl shrink-0 flex items-center justify-center font-bold text-sm border transition-colors ${isSelected ? 'bg-blue-600 text-white border-blue-600' : 'bg-slate-100 text-slate-500 border-slate-200'}">${letters[idx]}</span>
                    <span class="text-sm sm:text-base font-medium mt-1 ${isSelected ? 'text-blue-900 font-bold' : 'text-slate-700'}">${optText}</span>
                `;
                grid.appendChild(btn);
            });

            document.getElementById('btn-prev').disabled = currentQIndex === 0;
            document.getElementById('btn-next-text').innerText = currentQIndex === 9 ? "Завершить модуль" : "Далее";
            
            for (let i=0; i<10; i++) {
                const dot = document.getElementById(`dot-${i}`);
                if (dot) dot.className = `h-2 flex-grow rounded-full transition-colors ${i === currentQIndex ? 'bg-blue-600' : (userAnswers[i] !== undefined ? 'bg-blue-300' : 'bg-slate-200')}`;
            }
        }

        function prevQuestion() {
            if (autoAdvanceTimer) { clearTimeout(autoAdvanceTimer); autoAdvanceTimer = null; }
            if (currentQIndex > 0) { currentQIndex--; loadQuestion(); }
        }
        
        function nextQuestion() {
            if (autoAdvanceTimer) { clearTimeout(autoAdvanceTimer); autoAdvanceTimer = null; }
            if (currentQIndex < 9) { 
                currentQIndex++; loadQuestion(); 
            } else {
                const answered = Object.keys(userAnswers).length;
                if (answered < 10) {
                    if(!confirm(`Вы ответили на ${answered} из 10 вопросов. Все равно завершить модуль?`)) return;
                }
                finishModuleTest();
            }
        }

        function finishModuleTest() {
            if (timerInterval) clearInterval(timerInterval);
            
            let score = 0;
            testQuestions.forEach((q, i) => {
                if (userAnswers[i] === q.correct) score++;
            });

            // Save locally
            resultsStore[currentModuleId] = { score: score, total: 10 };
            localStorage.setItem('mob_results', JSON.stringify(resultsStore));

            // Render view
            document.getElementById('result-module-title').innerText = `Модуль ${currentModuleId}`;
            document.getElementById('result-score').innerText = score;
            
            const syncStatus = document.getElementById('result-sync-status');
            syncStatus.innerHTML = `<svg class="w-4 h-4 animate-spin text-blue-500" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path></svg>
                                    <span>Отправка данных в ведомость...</span>`;
            
            showView('view-module-result');
            sendToGoogleScript(currentModuleId, score, syncStatus);
        }

        function sendToGoogleScript(modId, score, statusEl) {
            if (!GOOGLE_SCRIPT_URL || GOOGLE_SCRIPT_URL.includes("ВСТАВЬТЕ_СЮДА")) {
                statusEl.innerHTML = `<svg class="w-4 h-4 text-amber-500" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M12 9v2m0 4h.01m-6.938 4h13.856c1.54 0 2.502-1.667 1.732-3L13.732 4c-.77-1.333-2.694-1.333-3.464 0L3.34 16c-.77 1.333.192 3 1.732 3z"></path></svg>
                                      <span class="text-amber-600">Сохранено локально (URL таблицы не настроен)</span>`;
                return;
            }

            const payload = {
                name: currentUser.name,
                phone: currentUser.phone,
                module: modId,
                score: score,
                total: 10
            };

            fetch(GOOGLE_SCRIPT_URL, {
                method: 'POST',
                headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                body: JSON.stringify(payload)
            })
            .then(response => {

                                      <span class="text-emerald-600">Успешно сохранено в Excel</span>`;
            })
            .catch(error => {
                statusEl.innerHTML = `<svg class="w-4 h-4 text-rose-500" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"></path></svg>

            });
        }

        function triggerCelebration() {
            var duration = 3 * 1000;
            var animationEnd = Date.now() + duration;
            var defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 100 };

            function randomInRange(min, max) {
              return Math.random() * (max - min) + min;
            }

            var interval = setInterval(function() {
              var timeLeft = animationEnd - Date.now();
              if (timeLeft <= 0) return clearInterval(interval);

              var particleCount = 50 * (timeLeft / duration);
              confetti(Object.assign({}, defaults, { particleCount, origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 } }));
              confetti(Object.assign({}, defaults, { particleCount, origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 } }));
            }, 250);
        }
